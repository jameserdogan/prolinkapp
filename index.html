        function toggleCheckbox(id) {

            const checkbox = document.getElementById(id);

            const item = checkbox.parentElement;

            

            checkbox.checked = !checkbox.checked;

            

            if (checkbox.checked) {

                item.classList.add('selected');

            } else {

                item.classList.remove('selected');

            }

        }

        function loadProfiles() {

            const cardStack = document.getElementById('cardStack');

            currentCardIndex = 0;

            updateCounter();

            

            // Add touch/mouse events to top card

            addSwipeEvents();

        function addSwipeEvents() {

            const topCard = document.querySelector('.profile-card');

            if (!topCard) return;

            

            let isDragging = false;

            let startX = 0;

            let currentX = 0;

            let cardRect = topCard.getBoundingClientRect();

            

            // Mouse events

            topCard.addEventListener('mousedown', startDrag);

            document.addEventListener('mousemove', drag);

            document.addEventListener('mouseup', endDrag);

            

            // Touch events

            topCard.addEventListener('touchstart', startDrag);

            document.addEventListener('touchmove', drag);

            document.addEventListener('touchend', endDrag);

            

            function startDrag(e) {

                isDragging = true;

                startX = e.type === 'mousedown' ? e.clientX : e.touches[0].clientX;

                topCard.classList.add('dragging');

                cardRect = topCard.getBoundingClientRect();

            }

            

            function drag(e) {

                if (!isDragging) return;

                e.preventDefault();

                

                currentX = (e.type === 'mousemove' ? e.clientX : e.touches[0].clientX) - startX;

                const rotation = currentX * 0.1;

                

                topCard.style.transform = `translateX(${currentX}px) rotate(${rotation}deg)`;

                

                // Visual feedback

                if (Math.abs(currentX) > 50) {

                    topCard.style.borderColor = currentX > 0 ? '#6dd4ec' : '#000000';

                    topCard.style.borderWidth = '3px';

                } else {

                    topCard.style.borderColor = 'transparent';

                }

            }

            

            function endDrag() {

                if (!isDragging) return;

                isDragging = false;

                

                topCard.classList.remove('dragging');

                topCard.style.borderColor = 'transparent';

                

                if (Math.abs(currentX) > 100) {

                    // Swipe detected

                    if (currentX > 0) {

                        swipeRight();

                    } else {

                        swipeLeft();

                    }

                } else {

                    // Snap back

                    topCard.style.transform = '';

                }

                

                currentX = 0;

            }

        }

        

        function swipeLeft() {

            const topCard = document.querySelector('.profile-card');

            if (!topCard) return;

            

            const userId = topCard.dataset.userId;

            recordSwipe(userId, false);

            

            topCard.classList.add('swiped-left');

            setTimeout(() => {

                nextCard();

            }, 300);

        }

        

        function swipeRight() {

            const topCard = document.querySelector('.profile-card');

            if (!topCard) return;

            

            const userId = topCard.dataset.userId;

            recordSwipe(userId, true);

            

            topCard.classList.add('swiped-right');

            setTimeout(() => {

                nextCard();

            }, 300);

        }

        

        function recordSwipe(userId, isLike) {

            if (!swipes[currentUser.id]) {

                swipes[currentUser.id] = {};

            }

            swipes[currentUser.id][userId] = isLike;

            

            // Check for match

            if (isLike && swipes[userId] && swipes[userId][currentUser.id]) {

                // It's a match!

                const matchedUser = attendees.find(a => a.id === userId);

                matches.push({

                    user1: currentUser,

                    user2: matchedUser,

                    timestamp: new Date()

                });

                

                showMatchNotification();

                updateMatches();

            }

            

            // Save to localStorage

            localStorage.setItem('prolink_swipes', JSON.stringify(swipes));

            localStorage.setItem('prolink_matches', JSON.stringify(matches));

        }

        

        function showMatchNotification() {

            const notification = document.getElementById('matchNotification');

            notification.style.display = 'block';

            setTimeout(() => {

                notification.style.display = 'none';

            }, 3000);

        }

        

        function nextCard() {

            currentCardIndex++;

            updateCounter();

            

            const cardStack = document.getElementById('cardStack');

            const cards = cardStack.querySelectorAll('.profile-card');

            

            // Remove the top card

            if (cards[0]) {

                cards[0].remove();

            }

            

            // Add new card to bottom if available

            if (currentCardIndex + 2 < profileCards.length) {

                const newCard = createProfileCard(profileCards[currentCardIndex + 2]);

                cardStack.innerHTML += newCard;

            }

            

            // Check if done

            if (currentCardIndex >= profileCards.length) {

                document.getElementById('noMoreCards').style.display = 'block';

                document.querySelector('.swipe-buttons').style.display = 'none';

                return;

            }

            

            // Add events to new top card

            setTimeout(addSwipeEvents, 50);

        }

        

        function updateCounter() {

            document.getElementById('currentCard').textContent = Math.min(currentCardIndex + 1, profileCards.length);

        }

        

        function updateMatches() {

            const matchesList = document.getElementById('matchesList');

            const matchesSection = document.getElementById('matchesSection');

            

            if (matches.length === 0) {

                matchesSection.style.display = 'none';

                return;

            }

            

            const userMatches = matches.filter(match => 

                match.user1.id === currentUser.id || match.user2.id === currentUser.id

            );

            

            if (userMatches.length > 0) {

                matchesList.innerHTML = userMatches.map(match => {

                    const otherUser = match.user1.id === currentUser.id ? match.user2 : match.user1;

                    return `

                        <div class="match-card">

                            <div class="match-name">${otherUser.name}</div>

                            <div class="match-details">${otherUser.business} • ${otherUser.industry}</div>

                            ${otherUser.email ? `<div class="match-details">📧 ${otherUser.email}</div>` : ''}

                        </div>

                    `;

                }).join('');

                

                matchesSection.style.display = 'block';

            }

        }
